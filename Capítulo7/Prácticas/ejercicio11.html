Código de JavaScript

var litros, l, kilometraje, km, lxkm, totalkm, totallt, viaje;
totalkm = 0;
totallt = 0;
viaje = 1;

kilometraje = window.prompt("Escriba los kilometros recorridos del viaje (enteros), sólo utilice números enteros y mayores a cero: ", "0");
km = parseInt(kilometraje);

while(km != 0){
totalkm += km;
litros = window.prompt("Escriba los litros consumidos del viaje (enteros):","0");
l = parseInt(litros)
lxkm = km / l;
totallt += l;
document.writeln("< p >En el viaje "+ viaje + " se obtuvo "+ lxkm +" kilometros recorridos por cada litro. ")
viaje++;
kilometraje = window.prompt("Escriba los kilometros recorridos del viaje (enteros), sólo utilice números enteros y mayores a cero: ", "0");
km = parseInt(kilometraje);
}
lxkm = totalkm / totallt;
document.writeln("< p >El total de kilometros recorridos por litro es "+ lxkm +"");
