var primerNumero, segundoNumero;
var numero1, numero2;

primerNumero = window.prompt("Escriba el primer entero: ");
segundoNumero = window.prompt("Escriba el segundo entero: ");

numero1 = parseInt(primerNumero);
numero2 = parseInt(segundoNumero);

if (numero1 % numero2 == 0){
document.writeln("< h1 > El numero " + numero1 + "es multiplo de" + numero2 + "");
} else{
document.writeln("< h1 > El numero " + numero1 + " no es multiplo de" + numero2 + "");
}
