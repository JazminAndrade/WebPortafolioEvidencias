var r;
var diametro;
var circunferencia;
var area;

r = window.prompt("Ingrese el valor del radio del círculo en cuestión:");
diametro = 2 * r;
circunferencia = 2 * Math.PI * r;
area = Math.PI * (r * r);
document.writeln("El diámetro del círculo es " + diametro + ", la circunferencia es " + circunferencia + " y el área es " + area);
